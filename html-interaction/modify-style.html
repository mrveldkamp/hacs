Modify Style

Topic Explanation

JavaScript may be used to select an element and modify its style.

Modify an element with the style property.

CSS properties with hyphens need to be changed to camelCasing.
(eg. font-size -> fontSize, font-family -> fontFamily)

index.html


<body>
<h1 id="heading">This is a Heading</h1>
<p id="para">This is a paragraph.</p>
<p><button id="btn">Click Me</button></p>

<!-- Run External JavaScript File -->
<script src="main.js"></script>
</body>

main.js


// Variables to store reference to HTML Elements
let btnEl = document.getElementById("btn");
let headingEl = document.getElementById("heading");
let paragraphEl = document.getElementById("para");

// Run modifyStyle function when button is clicked 
btnEl.addEventListener("click", modifyStyle);

function modifyStyle() {
  // Use the style property to modify element styles
  headingEl.style.color = "red";
  headingEl.style.fontFamily = "Comic Sans MS";
  paragraphEl.style.textDecoration = "underline";
  paragraphEl.style.textAlign = "center";
}
