Modify Style with JavaScript

JavaScript may be used to select an element and modify its style.

Modify an element with the style property:

document.getElementById("id").style.cssProperty = "cssValue";

CSS properties with hyphens need to be changed to camelCasing.
(eg. font-size -> fontSize, font-family -> fontFamily)

index.html


<body>
  <h1 id="headingId">This is a Heading</h1>
  <p id="paraId">This is a paragraph.</p>
  <p><button id="btnId">Click Me</button></p>
</body>

main.js


// Variables to store reference to HTML Elements
let btnEl = document.getElementById("btnId");
let headingEl = document.getElementById("headingId");
let paragraphEl = document.getElementById("paraId");

// Run modifyStyle function when button is clicked 
btnEl.addEventListener("click", modifyStyle);

function modifyStyle() {
  // Use the style property to modify element styles
  headingEl.style.color = "red";
  headingEl.style.fontFamily = "Comic Sans MS";
  paragraphEl.style.textDecoration = "underline";
  paragraphEl.style.textAlign = "center";
}

Modify Class with JavaScript

JavaScript may be used to select an element and access its class attribute.

Access the class of an element with the className property:

document.getElementById("id").className = "newCssClass";

index.html


<body>
  <p><button id="btnId" class="active">Click Me</button></p>
</body>

style.css


.active {
  text-decoration: underline red;
}

main.js


// Variables to store reference to HTML Elements
let btnEl = document.getElementById("btnId");

// Run modifyStyle function when button is clicked 
btnEl.addEventListener("click", toggleActive);

function toggleActive() {
  // Toggle Button Active Class when Button Clicked
  if (btnEl.className === "active") {
    btnEl.className = "";
  } else {
    btnEl.className = "active";
  }
}

Summary

  • Use the style property to modify the style of an element.
  • JavaScript does not support hyphens, so CSS properties with hyphens need to be changed to camelCasing.
  • Use the className property to access the class of an element.
